<script>
  document.addEventListener("DOMContentLoaded", function() {
    // Verifica se estamos na homepage
    if (window.location.pathname !== '/') {
      // Carrega o script de rolagem infinita
      var infiniteScrollScript = document.createElement('script');
      infiniteScrollScript.src = 'https://matthewrayfield.github.io/tumblr-infinite-scrolling/infinitescrolling.js'; // URL do seu script
      document.head.appendChild(infiniteScrollScript);

      // Função para mostrar posts com efeito suave
      function showPosts(posts) {
        posts.forEach((post, index) => {
          setTimeout(() => {
            post.classList.add('visible'); // Adiciona a classe 'visible' para cada post
          }, index * 50); // Atraso para cada post
        });
      }

      // Função para carregar novos posts
      function loadNewPosts() {
        const newPosts = document.querySelectorAll('.single-post:not(.visible)'); // Seleciona apenas os novos posts
        const postsToShow = Array.from(newPosts).slice(0, 15); // Pega os primeiros 15 novos posts
        if (postsToShow.length > 0) {
          showPosts(postsToShow); // Mostra os novos posts com efeito suave
        }
      }

      // Observer para detectar quando novos posts são carregados
      const observer = new MutationObserver(() => {
        loadNewPosts(); // Chama a função quando novos posts são adicionados
      });

      // Configura o observer para observar mudanças no DOM
      observer.observe(document.body, { childList: true, subtree: true });

      // Carregamento inicial dos posts
      const initialPosts = document.querySelectorAll('.single-post');
      showPosts(initialPosts);

      // Carregar novos posts de forma contínua a cada 5 segundos (opcional)
      setInterval(loadNewPosts, 3000);
    }
  });
</script>
